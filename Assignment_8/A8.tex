\section{\textbf{Problem}}
If A is an $m*n$ matrix, B is a $n*m$ matrix and $n<m$, then Prove that AB is not invertible. 
\section{\textbf{Solution}}
Let us represent A and B as follows
\begin{align}
     A = (a_{ij}) \in \Re^{m*n}\\
     B = (b_{ij}) \in \Re^{n*m}
\end{align}
Let C = AB which would be an $m*m$ matrix.
\begin{align}
    C = (c_{ij}) \in \Re^{m*m}\\
    c_{ij} = \sum_{k=1}^{n} a_{ik}b_{kj}
\end{align}
for j=1,2,....,m. Let 
\begin{align} 
    \vec{c}_{j} = C_{e{j}} \in \Re^{m}
\end{align}
where $(e_{1},e_{2},......,e_{m})$ stands for the canonical basis of $\Re^{m}$. Then, we have
\begin{align}
      c_{j} = \myvec{\sum_{k=1}^{n} a_{1k}b_{kj} \\.\\.\\.\\ \sum_{k=1}^{n} a_{mk}b_{kj}}\\
      = \sum_{k=1}^{n}b_{kj} \myvec{a_{1k} \\.\\.\\. a_{mk}}\\
      = \sum_{k=1}^{n}b_{kj}\vec{a}_{k}
\end{align}
with
\begin{align}
       \vec{a}_{j} = \myvec{a_{1j} \\.\\.\\ a_{mj}} \in \Re^{n}
\end{align}
That gives us,
\begin{align}
    \abs{C} = \abs{\vec{c}_{1},.....,\vec{c}_{m}}\\
    = \sum_{k_{1}=1}^{n}\sum_{k_{2}=1}^{n}..\sum_{k_{m}=1}^{n}b_{k_{1},1},b_{k_{2},2},..,b_{k_{m},m}\abs{\vec{a}_{k_{1}},..,\vec{a}_{k_{m}}}
\end{align}
Since there are exactly n vectors $\vec{a}_j$ and the determinant has $m>n$ entries, then we have
\begin{align}
    \abs{\vec{a}_{k_{1}},..,\vec{a}_{k_{m}}} = 0\\
    \forall k_{1},...k_{m} \in (1,2,..,n)
\end{align}
Because atleast two entries must be equal. Hence
\begin{align}
          \abs{C} = 0
\end{align}
Since $\abs{AB} = 0$. Therefore, AB is not invertible.
